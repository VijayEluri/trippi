61      public TrippiConsole(TrippiConfig config, 
62                           TrippiProfile profile,
63                           InputStream in) throws Exception {
64          m_config = config;
65          m_reader = new BufferedReader(new InputStreamReader(in));
66          System.out.println("Welcome to the Trippi v" + Trippi.VERSION + " Console.");
67          if (profile == null) {
68              System.out.println("You are not connected to a triplestore.");
69          } else {
70              connect(profile);
71          }
72          System.out.println("\nType 'help;' for help.\n");
73          boolean done = false;
74          while (!done) {
75              done = inputAndRun();
76          }
77          if (m_connector != null) {
78              m_connector.close();
79          }
80      }

95      private boolean inputAndRun() throws IOException {
96          String prompt;
97          if (m_profile == null) {
98              prompt = "trippi> ";
99          } else {
100             prompt = "trippi/" + m_profile.getId() + "> ";
101         }
102         System.out.print(prompt);
103         // input until trailing ; is reached
104         StringBuffer input = new StringBuffer();
105         String line;
106         boolean finished = false;
107         boolean done = false;
108         while (!done) {
109             line = m_reader.readLine();
110             if (line == null) {
111                 done = true; finished = true;
112             } else {
113                 line = line.trim();
114                 if (line.endsWith(";")) {
115                     input.append((line.substring(0, line.length() - 1)).trim());
116                     done = true;
117                 } else {
118                     input.append(line + " ");
119                     System.out.print("> ");
120                 }
121             }
122         }
123         if (finished) return true;
124         // now look at the command and interpret it
125         String cmdLine = input.toString().trim();
126         String u = cmdLine.toUpperCase();
127         if ( u.startsWith("EXIT") || u.startsWith("QUIT")) {
128             return true;
129         } else {
130             try {
131                 System.out.println("");
132                 String[] tokens = cmdLine.split(" ");
133                 if (tokens == null || tokens.length < 2) {
134                     interpret(cmdLine.toLowerCase(), "");
135                 } else {
136                     interpret(tokens[0].toLowerCase(), cmdLine.substring(tokens[0].length() + 1));
137                 }
138             } catch (Exception e) {
139                 System.out.println("ERROR: " + e.getMessage());
140                 if (m_trace) {
141                     printFullTrace(e);
142                 }
143             } finally {
144                 System.out.println("");
145             }
146             return false;
147         }
148     }

150     private void printFullTrace(Throwable e) {
151         int level = 0;
152         while (e != null) {
153             if (level > 0) System.out.println("CAUSED BY:");
154             level++;
155             e.printStackTrace();
156             e = e.getCause();
157         }

160     public void doDateTest(String cmd) throws Exception {
161         if (cmd == null || cmd.equals("")) {
162             System.out.println("Usage: datetest [plain|double|dateTime] [numTriples]");
163         } else {
164             String[] parts = cmd.split(" ");
165             if (parts.length == 2) {
166                 int num = Integer.parseInt(parts[1]); 
167                 int type;
168                 if (parts[0].equals("plain")) {
169                     type = 0;
170                 } else if (parts[0].equals("double")) {
171                     type = 1;
172                 } else if (parts[0].equals("dateTime")) {
173                     type = 2;
174                 } else {
175                     throw new TrippiException("Unrecognized datetest type: " + parts[0]);
176                 }
177                 System.out.print("Adding " + num + " <resource:NUM> <urn:date> (" + parts[0] + ") triples...");
178                 long startTime = System.currentTimeMillis();
179                 String property = "urn:date";
180                 SimpleDateFormat formatter = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
181                 for (int i = 0; i < num; i++) {
182                     String resource = "resource:" + i;
183                     Triple triple;
184                     if (type == 1) {
185                         triple = TripleMaker.createTyped(resource,
186                                                            property,
187                                                            "" + i,
188                                                            "http://www.w3.org/2001/XMLSchema#double");
189                     } else {
190                         long epochMS = (long) (i * 1000);
191                         Date date = new Date();
192                         date.setTime(epochMS);
193                         String dateString = formatter.format(date);
194                         if (type == 0) {
195                             triple = TripleMaker.createPlain(resource,
196                                                                property,
197                                                                dateString);
198                         } else {
199                             triple = TripleMaker.createTyped(resource,
200                                                                property,
201                                                                dateString,
202                                                                "http://www.w3.org/2001/XMLSchema#dateTime");
203                         }
204                     }
205                     m_connector.getWriter().add(triple, false);
206                 }
207                 System.out.println("OK.");
208                 long endTime = System.currentTimeMillis();
209                 double elapsed = ((double) (endTime - startTime)) / 1000.0;
210                 System.out.println(elapsed + " seconds elapsed.");
211             } else {
212                 System.out.println("Usage: datetest [plain|double|dateTime] [numTriples]");
213             }
214         }
215     }

217     public void interpret(String cmd, String rest) throws Exception {
218         if (cmd.equals("help")) {
219             doHelp();
220         } else if (cmd.equals("profiles")) {
221             doProfiles();
222         } else if (cmd.equals("connectors")) {
223             doConnectors();
224         } else if (cmd.equals("use")) {
225             doUse(rest);
226         } else if (cmd.equals("create")) {
227             doCreate();
228         } else if (cmd.equals("alias")) {
229             doAlias(rest);
230         } else if (cmd.equals("datetest")) {
231             checkProfile();
232             doDateTest(rest);
233         } else if (cmd.equals("add")) {
234             checkProfile();
235             doAdd(rest);
236         } else if (cmd.equals("delete")) {
237             checkProfile();
238             doDelete(rest);
239         } else if (cmd.equals("tuples")) {
240             checkProfile();
241             doTuples(rest, false);
242         } else if (cmd.equals("count")) {
243             checkProfile();
244             if (rest.startsWith("tuples")) {
245                 doTuples(rest.substring(7), true);
246             } else if (rest.startsWith("triples")) {
247                 doTriples(rest.substring(8), true);
248             } else if (rest.equals("")) {
249                 doTriples("spo * * *", true);
250             } else {
251                 throw new TrippiException("Bad syntax, try count by itself or count [tuples ..|triples ..]");
252             }
253         } else if (cmd.equals("triples")) {
254             checkProfile();
255             doTriples(rest, false);
256         } else if (cmd.equals("dump")) {
257             checkProfile();
258             doDump(rest);
259         } else if (cmd.startsWith("triplef")) {
260             if (rest.equals("")) {
261                 System.out.println("Current Triple Result Format: " + m_tripleForm.getName());
262                 System.out.print("Formats Supported: ");
263                 RDFFormat[] p = TripleIterator.OUTPUT_FORMATS;
264                 for (int i = 0; i < p.length; i++) {
265                     if (i > 0) System.out.print(", ");
266                     System.out.print(p[i].getName());
267                 }
268                 System.out.println("");
269             } else {
270                 m_tripleForm = RDFFormat.forName(rest);
271                 System.out.println("New Triple Result Format: " + m_tripleForm.getName());
272             }
273         } else if (cmd.startsWith("tuplef")) {
274             if (rest.equals("")) {
275                 System.out.println("Current Tuple Result Format: " + m_tupleForm.getName());
276                 System.out.print("Formats Supported: ");
277                 RDFFormat[] p = TupleIterator.OUTPUT_FORMATS;
278                 for (int i = 0; i < p.length; i++) {
279                     if (i > 0) System.out.print(", ");
280                     System.out.print(p[i].getName());
281                 }
282                 System.out.println("");
283             } else {
284                 m_tupleForm = RDFFormat.forName(rest);
285                 System.out.println("New Tuple Result Format: " + m_tupleForm.getName());
286             }
287         } else if (cmd.equals("load")) {
288             checkProfile();
289             doLoad(rest);
290         } else if (cmd.equals("close")) {
291             checkProfile();
292             System.out.print("Closing...");
293             m_connector.close();
294             System.out.println("OK.");
295             m_connector = null;
296             m_profile = null;
297         } else if (cmd.equals("limit")) {
298             doLimit(rest);
299         } else if (cmd.equals("distinct")) {
300             if (m_distinct) {
301                 m_distinct = false;
302                 System.out.println("Forced distinct mode is off.");
303             } else {
304                 m_distinct = true;
305                 System.out.println("Forced distinct mode is on.");
306             }
307         } else if (cmd.equals("trace")) {
308             if (m_trace) {
309                 m_trace = false;
310                 System.out.println("Tracing is disabled.");
311             } else {
312                 m_trace = true;
313                 System.out.println("Tracing is enabled.");
314             }
315         } else {
316             throw new TrippiException("Unrecognized command: " + cmd);
317         }
318     }

320     private static String doAliasReplacements(String string, Map<String, String> aliasMap) {
321         String out = string;
322         Iterator<String> iter = aliasMap.keySet().iterator();
323         while (iter.hasNext()) {
324             String alias = iter.next();
325             String fullForm = aliasMap.get(alias);
326             out = out.replaceAll("<" + alias + ":", "<" + fullForm);
327             out = out.replaceAll("\\^\\^" + alias + ":", "^^" + fullForm);
328         }
329         if (!string.equals(out)) {
330             logger.info("Substituted aliases, string is now: " + out);
331         }
332         return out;
333     }

384     public void doAlias(String rest) throws Exception {
385         Map<String, String> aliases = m_config.getAliasMap();
386         if (rest.equals("")) {
387             // show value of all aliases
388             Iterator<String> iter = aliases.keySet().iterator();
389             while (iter.hasNext()) {
390                 String alias = iter.next();
391                 String value = aliases.get(alias);
392                 System.out.println(alias + "\t" + value);
393             }
394         } else if (rest.equals("!")) {
395             // clear all aliases
396             Map<String, String> emptyMap = new HashMap<String, String>();
397             m_config.setAliasMap(emptyMap);
398             if (m_connector != null) 
399                     m_connector.getReader().setAliasMap(emptyMap);
400             System.out.println("Cleared all aliases.");
401         } else if (rest.indexOf(" ") == -1) {
402             // show value of one alias
403             String val = (String) aliases.get(rest);
404             if (val == null) {
405                 throw new TrippiException("No such alias: " + rest);
406             } else {
407                 System.out.println(rest + "\t" + val);
408             }
409         } else {
410             String[] tokens = rest.split(" ");
411             if (tokens.length != 2) {
412             } else if (tokens[1].equals("!")) {
413                 // clear one alias
414                 aliases.remove(tokens[0]);
415                 m_config.setAliasMap(aliases);
416                 if (m_connector != null) 
417                         m_connector.getReader().setAliasMap(aliases);
418                 System.out.println("Cleared alias: " + tokens[0]);
419             } else {
420                 // set an alias
421                 aliases.put(tokens[0], tokens[1]);
422                 m_config.setAliasMap(aliases);
423                 if (m_connector != null) 
424                         m_connector.getReader().setAliasMap(aliases);
425                 System.out.println(tokens[0] + " is now an alias for " + tokens[1]);
426             }
427         }
428     }

457     public void doTuples(String parms, boolean count) throws Exception {
458         // parse it first
459         String[] tokens = parms.split(" ");
460         if (tokens.length < 2) {
461             throw new TrippiException("No query specified.");
462         }
463         String lang = tokens[0];
464         String query = parms.substring(parms.indexOf(" ")).trim();
465         TupleIterator iter = null;
466         try {
467             long startTime = System.currentTimeMillis();
468             iter = m_connector.getReader().findTuples(lang, query, m_limit, m_distinct);
469             int n = 0;
470             if (count) {
471                 n = iter.count();
472             } else {
473                 iter.setAliasMap(m_connector.getReader().getAliasMap());
474                 n = iter.toStream(System.out, m_tupleForm);
475             }
476             System.out.println("Total Tuples   : " + n);
477             long endTime = System.currentTimeMillis();
478             double elapsedSeconds = ((double) endTime - startTime) / 1000.0;
479             System.out.println("Seconds Elapsed: " + elapsedSeconds);
480         } finally {
481             if (iter != null) iter.close();
482         }
483     }

485     public void doTriples(String parms, boolean count) throws Exception {
486         TripleIterator iter = null;
487         try {
488             int rightPos = parms.indexOf(")");
489             long startTime = System.currentTimeMillis();
490             if (parms.startsWith("(") && rightPos != -1) {
491                 // do a tuples-to-triples query
492                 String pattern = parms.substring(0, rightPos);
493                 String rest = parms.substring(rightPos+1).trim();
494                 String[] t = rest.split(" ");
495                 if (t.length < 2) {
496                     throw new TrippiException("Bad syntax, need lang and query.");
497                 }
498                 String lang = t[0];
499                 String query = rest.substring(rest.indexOf(" ")).trim();
500                 iter = m_connector.getReader().findTriples(lang, query, pattern, m_limit, m_distinct);
501             } else {
502                 // do a real triple query (lang query)
503                 String[] tokens = parms.split(" ");
504                 if (tokens.length < 2) {
505                     throw new TrippiException("Triple query must have at least two parameters.");
506                 }
507                 String lang = tokens[0];
508                 String query = parms.substring(parms.indexOf(" ")).trim();
509                 iter = m_connector.getReader().findTriples(lang, query, m_limit, m_distinct);
510             }
511             doTriples(iter, count, startTime);
512         } finally {
513             if (iter != null) iter.close();
514         }
515

533     public void doCreate() throws Exception {
534         System.out.println("Creating a new profile. To cancel at any time, enter CANCEL.\n");
535         boolean done = false;
536         boolean canceled = false;
537         String className = null;
538         ConnectorDescriptor descriptor = null;
539         while (!done) {
540             System.out.print("Enter the connector class name: ");
541             className = m_reader.readLine();
542             if (className == null || className.equalsIgnoreCase("CANCEL")) {
543                 canceled = true; done = true;
544             } else if (!className.equals("")) {
545                 try {
546                     descriptor = TriplestoreConnector.getDescriptor(className);
547                     done = true;
548                 } catch (TrippiException e) {
549                     System.out.println("ERROR: " + e.getMessage());
550                 }
551             }
552         }
553         if (canceled) {
554             System.out.println("Canceled.");
555         } else {
556             String id = null;
557             done = false; canceled = false;
558             while (!done) {
559                 System.out.print("Enter a short id for the new profile: ");
560                 id = m_reader.readLine();
561                 if (id == null || id.equalsIgnoreCase("CANCEL")) {
562                     canceled = true; done = true;
563                 } else if (!id.equals("")) {
564                     TrippiProfile p = (TrippiProfile) m_config.getProfiles().get(id);
565                     if (p != null) {
566                         System.out.println("ERROR: A profile with that id already exists.");
567                     } else {
568                         done = true;
569                     }
570                 }
571             }
572             if (canceled) {
573                 System.out.println("Canceled.");
574             } else {
575                 String label = null;
576                 done = false; canceled = false;
577                 while (!done) {
578                     System.out.print("Enter a label for the new profile: ");
579                     label = m_reader.readLine();
580                     if (label == null || label.equalsIgnoreCase("CANCEL")) {
581                         canceled = true; done = true;
582                     } else if (!label.equals("")) {
583                         done = true;
584                     }
585                 }
586                 if (canceled) {
587                     System.out.println("Canceled.");
588                 } else {
589                     TrippiProfile profile = configure(id, label, className, descriptor);
590                     if (profile != null) {
591                         Map<String, TrippiProfile> profileMap = m_config.getProfiles();
592                         profileMap.put(id, profile);
593                         m_config.setProfiles(profileMap);
594                         System.out.println("\nProfile created.  Type 'use " + id + ";' to use it.");
595                     } else {
596                         System.out.println("Canceled.");
597                     }
598                 }
599             }
600         }
601     }

625     private Map<String, String> configure(ConnectorParameter param) throws IOException {
626         Map<String, String> map = new HashMap<String, String>();
627         System.out.println("");
628         System.out.println("Parameter   : " + param.getName());
629         String desc      = "Description : ";
630         String blank     = "              ";
631         System.out.println(desc + formatString(param.getLabel(),
632                                                79 - desc.length(),
633                                                79,
634                                                desc.length()));
635         if (param.getDescription() != null && !param.getDescription().equals("")) {
636             System.out.println(blank + formatString(param.getDescription(),
637                                                     79 - desc.length(),
638                                                     79,
639                                                     desc.length()));
640         }
641         System.out.println("");
642         List<String> options = param.getOptions();
643         if (options.size() == 0) {
644             boolean done = false;
645             String input = null;
646             while (!done) {
647                 System.out.print("Enter a value");
648                 if (param.isOptional()) {
649                     System.out.print(" (or [ENTER] for none)");
650                 }
651                 System.out.print(": ");
652                 input = m_reader.readLine();
653                 if (input == null || input.equalsIgnoreCase("CANCEL")) return null;
654                 if (input.equals("") && param.isOptional()) {
655                     done = true;
656                 }
657                 if (!input.equals("")) {
658                     done = true;
659                 }
660             }
661             if (!input.equals("")) map.put(param.getName(), input);
662         } else {
663             HashMap<String, List<ConnectorParameter>> o = new HashMap<String, List<ConnectorParameter>>();
664             Iterator<String> iter = options.iterator();
665             StringBuffer optString = new StringBuffer();
666             int i = 0;
667             while (iter.hasNext()) {
668                 i++;
669                 String val = iter.next();
670                 o.put(val, param.getParameters(val));
671                 if (i == options.size()) {
672                     optString.append(" or ");
673                 } else if (i > 1) {
674                     optString.append(", ");
675                 }
676                 optString.append(val);
677             }
678             boolean done = false;
679             String input = null;
680             while (!done) {
681                 System.out.print("Enter " + optString);
682                 if (param.isOptional()) {
683                     System.out.print(" (or [ENTER] for none)");
684                 }
685                 System.out.print(": ");
686                 input = m_reader.readLine();
687                 if (input == null || input.equalsIgnoreCase("CANCEL")) return null;
688                 if (input.equals("") && param.isOptional()) {
689                     done = true;
690                 } else if (o.containsKey(input)) {
691                     done = true;
692                 }
693             }
694             if (!input.equals("")) {
695                 // input was ok, so set value in map and move on
696                 map.put(param.getName(), input);
697                 List<ConnectorParameter> subs = o.get(input);
698                 if (subs != null) {
699                     Map<String, String> subMap = configure(subs);
700                     if (subMap == null) return null;
701                     map.putAll(subMap);
702                 }
703             }
704         }
705         return map;
706     }

786     public static String formatString(String in, int firstMax, int otherMax, int otherIndent) {
787         StringBuffer out = new StringBuffer();
788         String[] tokens = in.split(" ");
789         int charsLeft = firstMax;
790         int i = 0;
791         while ( i < tokens.length ) {
792             String token = tokens[i++];
793             out.append(token);
794             charsLeft -= token.length();
795             if (i < tokens.length && tokens[i].length() >= charsLeft) {
796                 out.append("\n" + indent(otherIndent));
797                 charsLeft = otherMax - otherIndent;
798             } else {
799                 out.append(' ');
800                 charsLeft--;
801             }
802         }
803         return out.toString();
804     }

806     public static String indent(int num) {
807         StringBuffer out = new StringBuffer();
808         for (int i = 0; i < num; i++) out.append(' ');
809         return out.toString();
810     }
