24   * Writes triples as JSON.
25   *
26   * The N2/Talis spec: http://n2.talis.com/wiki/RDF_JSON_Specification
27   * The W3C SPARQL/JSON spec: http://www.w3.org/TR/rdf-sparql-json-res/

61      public int write(TripleIterator iter) throws TrippiException {
62  
63              m_out.println("{\"results\":[");
64              int count = 0;
65              while (iter.hasNext()) {
66              	if (count > 0) m_out.print(',');
67              	Triple result = iter.next();
68                  m_out.println(toJSON(result));
69                  count++;
70              }
71              m_out.print("]}");
72  
73              m_out.flush();
74              iter.close();
75              return count;
76      }

78      private final static String JSON_TEMPLATE = "{\"%\" : {\"%\" : [ % ] } }";
79      
80      /**
81       * http://n2.talis.com/wiki/RDF_JSON_Specification
82       * http://docs.api.talis.com/platform-api/output-types/rdf-json
83       * @param triple
84       * @return JSON serialization of the triple
85       */
86      public static String toJSON(Triple triple) {
87      	String [] parts = JSON_TEMPLATE.split("%");
88      	String subject = triple.getSubject().stringValue();
89      	String predicate = triple.getPredicate().stringValue();
90      	String object = toJSON(triple.getObject());
91      	StringBuffer result = new StringBuffer((JSON_TEMPLATE.length() - 3) + subject.length() + predicate.length() + object.length());
92      	result.append(parts[0]);
93      	result.append(subject);
94      	result.append(parts[1]);
95      	result.append(predicate);
96      	result.append(parts[2]);
97      	result.append(object);
98      	result.append(parts[3]);
99      	return result.toString();
100     }

102     private static String toJSON(ObjectNode node){
103     	StringBuffer result = new StringBuffer();
104     	result.append('{');
105     	result.append("\"type\":");
106     	if (node.isLiteral()){
107     		result.append("\"literal\"");
108     	}
109     	else if(node.isURIReference()){
110     		result.append("\"uri\"");
111     	}
112     	else {
113     		result.append("\"bnode\"");
114     	}
115     	result.append(", \"value\":");
116     	result.append('"');
117     	result.append(node.stringValue().replaceAll("\"", "\\\\\""));
118     	result.append('"');
119     	if(node.isLiteral()){
120         	result.append(", \"datatype\":");
121         	result.append('"');
122     		((Literal)node).getDatatypeURI();
123         	result.append('"');
124     	}
125     	result.append('}');
126     	return result.toString();
127     }

129     public String getValue(Node node) {
130         String fullString = RDFUtil.toString(node);
131         if (m_aliases != null) {
132             if (node instanceof URIReference) {
133                 Iterator<Alias> iter = m_aliases.getAliases().values().iterator();
134                 while (iter.hasNext()) {
135                     Alias alias = iter.next();
136                     String prefix = alias.getKey();
137                     String expansion = alias.getExpansion();
138                     if (fullString.startsWith("<".concat(expansion))) {
139                         return fix("<" + prefix + ":" + fullString.substring(expansion.length() + 1));
140                     }
141                 }
142             } else if (node instanceof Literal) {
143                 Literal literal = (Literal) node;
144                 if (literal.getDatatypeURI() != null) {
145                     String uri = literal.getDatatypeURI().toString();
146                     Iterator<Alias> iter = m_aliases.getAliases().values().iterator();
147                     while (iter.hasNext()) {
148                         Alias a = iter.next();
149                         String alias = a.getKey();
150                         String expansion = a.getExpansion();
151                         if (uri.startsWith(expansion)) {
152                             StringBuffer out = new StringBuffer();
153                             out.append('"');
154                             out.append(literal.getLexicalForm().replaceAll("\"", "\\\""));
155                             out.append("\"^^");
156                             out.append(alias);
157                             out.append(':');
158                             out.append(uri.substring(expansion.length()));
159                             return fix(out.toString());
160                         }
161                     }
162                 }
163             }
164         }
165         return fix(fullString);
166     }
167 
168     private String fix(String in) {
169         if (in.charAt(0) == '"') {
170             // literal
171             String lit = in.substring(1, in.lastIndexOf('"'));
172             return lit.replaceAll("\\\\\"", "\"").replaceAll("\n", " ");
173         } else if (in.charAt(0) == '<') {
174             // resource
175             return in.substring(1, in.length() - 1);
176         } else {
177             return in;
178         }
179     }
