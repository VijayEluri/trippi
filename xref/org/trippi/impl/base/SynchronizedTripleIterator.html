7   /**
8    * Wraps a <code>TripleIterator</code> and automatically releases the
9    * current thread's lock on the associated SynchronizedTriplestoreSession
10   * when closed.
11   *
12   * @author cwilper@cs.cornell.edu
13   */
14  public class SynchronizedTripleIterator extends TripleIterator {
15  
16      private TripleIterator m_iter;
17      private SynchronizedTriplestoreSession m_session;
18      private boolean m_closed = false;
19  
20      public SynchronizedTripleIterator(TripleIterator iter,
21                                        SynchronizedTriplestoreSession session) {
22          m_iter = iter;
23          m_session = session;
24      }
25  
26      @Override
27  	public boolean hasNext() throws TrippiException {
28          boolean has = m_iter.hasNext();
29          if (!has) close(); // proactively
30          return has;
31      }
32      
33      @Override
34  	public Triple next() throws TrippiException {
35          return m_iter.next();
36      }
37  
38      /**
39       * Close the wrapped iterator and release the session to the pool.
40       */
41      @Override
42  	public void close() throws TrippiException {
43          if (!m_closed) {
44              // This ensures that even if the wrapped iter throws an exception,
45              // the thread's lock on the session is released.
46              try {
47                  m_iter.close();
48              } catch (TrippiException e) {
49                  throw e;
50              } finally {
51                  m_session.releaseLock();
52                  m_closed = true;
53              }
54          }
55      }
56  
57      /**
58       * Ensure close() gets called at garbage collection time.
59       */
60      @Override
61  	public void finalize() throws TrippiException {
62          close();
63      }

65  }
