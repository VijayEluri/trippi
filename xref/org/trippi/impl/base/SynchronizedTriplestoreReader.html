79      private String doAliasReplacements(String q) {
80          String out = q;
81          Map<String, String> m = m_aliasManager.getAliasMap();
82          Iterator<String> iter = m.keySet().iterator();
83          while (iter.hasNext()) {
84              String alias = iter.next();
85              String fullForm = m.get(alias);
86              out = out.replaceAll("<" + alias + ":", "<" + fullForm)
87                       .replaceAll("\\^\\^" + alias + ":(\\S+)", "^^<" + fullForm + "$1>");
88          }
89          if (!q.equals(out)) {
90              logger.info("Substituted aliases, query is now: " + out);
91          }
92          return out;
93      }

102     /**
103      * Delegates to underlying session for non-basic queries.
104      */
105     public TripleIterator findTriples(String queryLang,
106                                       String tripleQuery,
107                                       int limit,
108                                       boolean distinct) throws TrippiException {
109         if (queryLang.equals("spo")) {
110             // parse and call findTriples(subj, pred, obj)
111             TriplePattern[] patterns = TriplePattern.parse(doAliasReplacements(tripleQuery));
112             if (patterns.length == 1) {
113                 TriplePattern p = patterns[0];
114                 SubjectNode subject = null;
115                 PredicateNode predicate = null;
116                 ObjectNode object = null;
117                 if (p.getSubject() instanceof SubjectNode) subject = (SubjectNode) p.getSubject();
118                 if (p.getPredicate() instanceof PredicateNode) predicate = (PredicateNode) p.getPredicate();
119                 if (p.getObject() instanceof ObjectNode) object = (ObjectNode) p.getObject();
120                 return findTriples(subject, predicate, object, limit);
121             } else {
122                 throw new TrippiException("Only one triple pattern may be specified.");
123             }
124         } else {
125             // delegate to the session
126             TripleIterator iter = null;
127             iter = m_session.findTriples(queryLang, tripleQuery);
128             if (distinct) iter = new DistinctTripleIterator(iter);
129             if (limit > 0) iter = new LimitedTripleIterator(iter, limit);
130             return iter;
131         }
132     }

161                                       String tripleTemplate,
162                                       int limit,
163                                       boolean distinct) throws TrippiException {
164         TriplePattern[] patterns = TriplePattern.parse(tripleTemplate);
165         TripleIterator iter = new TupleBasedTripleIterator(findTuples(queryLang, 
166                                                                       tupleQuery,
167                                                                       -1,
168                                                                       distinct),
169                                                            patterns);
170         if (distinct) iter = new DistinctTripleIterator(iter);
171         if (limit > 0) iter = new LimitedTripleIterator(iter, limit);
