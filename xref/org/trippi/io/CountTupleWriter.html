17  public class CountTupleWriter extends TupleWriter {

23  	}

25  	public CountTupleWriter(Writer out) {
26  		_out = out;
27  	}
28  
29      @Override
30  	public int write(TupleIterator iter) throws TrippiException {
31          try {
32              /* Use the in-built TupleIterator.count() method */
33              int count = iter.count();
34              
35              /* Finger's crossed... */            
36              String str_count = Integer.toString(count);
37              
38              _out.write( str_count );
39              
40              /* Flush to OutputStream */
41              _out.flush();
42              
43              /* Documentation states that the TripleWriter should close the 
44                 Iterator, but not the stream, hence: */
45              iter.close();
46              return count;
47          } catch (IOException e) {
48              throw new TrippiException("Error setting up CountTupleWriter", e);
49          } 
50      }
