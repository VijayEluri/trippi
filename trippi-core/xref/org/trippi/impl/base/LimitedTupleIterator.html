9   /**
10   * Ensures that no more than a certain number of items are returned 
11   * by the wrapped iterator.
12   *
13   * @author cwilper@cs.cornell.edu
14   */
15  public class LimitedTupleIterator extends TupleIterator {
16  
17      private TupleIterator m_wrapped;
18      private int m_count;
19      private int m_limit;
20      private Map<String, Node> m_next;
21      private boolean m_closed = false;
22  
23      public LimitedTupleIterator(TupleIterator wrapped,
24                                  int limit) throws TrippiException {
25          m_wrapped = wrapped;
26          m_count = 0;
27          m_limit = limit;
28          m_next = getNext();
29      }
30  
31      // return null if there are no more or limit has been reached
32      private Map<String, Node> getNext() throws TrippiException {
33          if (m_wrapped.hasNext() && m_count < m_limit) {
34              m_count++;
35              return m_wrapped.next();
36          } else {
37              return null;
38          }
39      }
40  
41      @Override
42  	public boolean hasNext() {
43          return (m_next != null);
44      }
45  
46      @Override
47  	public Map<String, Node> next() throws TrippiException {
48          if (m_next == null) return null;
49          Map<String, Node> last = m_next;
50          m_next = getNext();
51          return last;
52      }
53  
54      @Override
55  	public String[] names() throws TrippiException {
56          return m_wrapped.names();
57      }
58  
59      @Override
60  	public void close() throws TrippiException {
61          if (!m_closed) {
62              m_wrapped.close();
63              m_closed = true;
64          }
65      }
66  
67      @Override
68  	public void finalize() throws TrippiException {
69          close();
70      }
