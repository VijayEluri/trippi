62      public TrippiConsole(TrippiConfig config, 
63                           TrippiProfile profile,
64                           InputStream in) throws Exception {
65          m_config = config;
66          m_reader = new BufferedReader(new InputStreamReader(in));
67          System.out.println("Welcome to the Trippi v" + Trippi.VERSION + " Console.");
68          if (profile == null) {
69              System.out.println("You are not connected to a triplestore.");
70          } else {
71              connect(profile);
72          }
73          System.out.println("\nType 'help;' for help.\n");
74          boolean done = false;
75          while (!done) {
76              done = inputAndRun();
77          }
78          if (m_connector != null) {
79              m_connector.close();
80          }
81      }

96      private boolean inputAndRun() throws IOException {
97          String prompt;
98          if (m_profile == null) {
99              prompt = "trippi> ";
100         } else {
101             prompt = "trippi/" + m_profile.getId() + "> ";
102         }
103         System.out.print(prompt);
104         // input until trailing ; is reached
105         StringBuffer input = new StringBuffer();
106         String line;
107         boolean finished = false;
108         boolean done = false;
109         while (!done) {
110             line = m_reader.readLine();
111             if (line == null) {
112                 done = true; finished = true;
113             } else {
114                 line = line.trim();
115                 if (line.endsWith(";")) {
116                     input.append((line.substring(0, line.length() - 1)).trim());
117                     done = true;
118                 } else {
119                     input.append(line + " ");
120                     System.out.print("> ");
121                 }
122             }
123         }
124         if (finished) return true;
125         // now look at the command and interpret it
126         String cmdLine = input.toString().trim();
127         String u = cmdLine.toUpperCase();
128         if ( u.startsWith("EXIT") || u.startsWith("QUIT")) {
129             return true;
130         } else {
131             try {
132                 System.out.println("");
133                 String[] tokens = cmdLine.split(" ");
134                 if (tokens == null || tokens.length < 2) {
135                     interpret(cmdLine.toLowerCase(), "");
136                 } else {
137                     interpret(tokens[0].toLowerCase(), cmdLine.substring(tokens[0].length() + 1));
138                 }
139             } catch (Exception e) {
140                 System.out.println("ERROR: " + e.getMessage());
141                 if (m_trace) {
142                     printFullTrace(e);
143                 }
144             } finally {
145                 System.out.println("");
146             }
147             return false;
148         }
149     }

151     private void printFullTrace(Throwable e) {
152         int level = 0;
153         while (e != null) {
154             if (level > 0) System.out.println("CAUSED BY:");
155             level++;
156             e.printStackTrace();
157             e = e.getCause();
158         }

161     public void doDateTest(String cmd) throws Exception {
162         if (cmd == null || cmd.equals("")) {
163             System.out.println("Usage: datetest [plain|double|dateTime] [numTriples]");
164         } else {
165             String[] parts = cmd.split(" ");
166             if (parts.length == 2) {
167                 int num = Integer.parseInt(parts[1]); 
168                 int type;
169                 if (parts[0].equals("plain")) {
170                     type = 0;
171                 } else if (parts[0].equals("double")) {
172                     type = 1;
173                 } else if (parts[0].equals("dateTime")) {
174                     type = 2;
175                 } else {
176                     throw new TrippiException("Unrecognized datetest type: " + parts[0]);
177                 }
178                 System.out.print("Adding " + num + " <resource:NUM> <urn:date> (" + parts[0] + ") triples...");
179                 long startTime = System.currentTimeMillis();
180                 String property = "urn:date";
181                 SimpleDateFormat formatter = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
182                 for (int i = 0; i < num; i++) {
183                     String resource = "resource:" + i;
184                     Triple triple;
185                     if (type == 1) {
186                         triple = TripleMaker.createTyped(resource,
187                                                            property,
188                                                            "" + i,
189                                                            "http://www.w3.org/2001/XMLSchema#double");
190                     } else {
191                         long epochMS = (long) (i * 1000);
192                         Date date = new Date();
193                         date.setTime(epochMS);
194                         String dateString = formatter.format(date);
195                         if (type == 0) {
196                             triple = TripleMaker.createPlain(resource,
197                                                                property,
198                                                                dateString);
199                         } else {
200                             triple = TripleMaker.createTyped(resource,
201                                                                property,
202                                                                dateString,
203                                                                "http://www.w3.org/2001/XMLSchema#dateTime");
204                         }
205                     }
206                     m_connector.getWriter().add(triple, false);
207                 }
208                 System.out.println("OK.");
209                 long endTime = System.currentTimeMillis();
210                 double elapsed = ((double) (endTime - startTime)) / 1000.0;
211                 System.out.println(elapsed + " seconds elapsed.");
212             } else {
213                 System.out.println("Usage: datetest [plain|double|dateTime] [numTriples]");
214             }
215         }
216     }

218     public void interpret(String cmd, String rest) throws Exception {
219         if (cmd.equals("help")) {
220             doHelp();
221         } else if (cmd.equals("profiles")) {
222             doProfiles();
223         } else if (cmd.equals("connectors")) {
224             doConnectors();
225         } else if (cmd.equals("use")) {
226             doUse(rest);
227         } else if (cmd.equals("create")) {
228             doCreate();
229         } else if (cmd.equals("alias")) {
230             doAlias(rest);
231         } else if (cmd.equals("datetest")) {
232             checkProfile();
233             doDateTest(rest);
234         } else if (cmd.equals("add")) {
235             checkProfile();
236             doAdd(rest);
237         } else if (cmd.equals("delete")) {
238             checkProfile();
239             doDelete(rest);
240         } else if (cmd.equals("tuples")) {
241             checkProfile();
242             doTuples(rest, false);
243         } else if (cmd.equals("count")) {
244             checkProfile();
245             if (rest.startsWith("tuples")) {
246                 doTuples(rest.substring(7), true);
247             } else if (rest.startsWith("triples")) {
248                 doTriples(rest.substring(8), true);
249             } else if (rest.equals("")) {
250                 doTriples("spo * * *", true);
251             } else {
252                 throw new TrippiException("Bad syntax, try count by itself or count [tuples ..|triples ..]");
253             }
254         } else if (cmd.equals("triples")) {
255             checkProfile();
256             doTriples(rest, false);
257         } else if (cmd.equals("dump")) {
258             checkProfile();
259             doDump(rest);
260         } else if (cmd.startsWith("triplef")) {
261             if (rest.equals("")) {
262                 System.out.println("Current Triple Result Format: " + m_tripleForm.getName());
263                 System.out.print("Formats Supported: ");
264                 RDFFormat[] p = TripleIterator.OUTPUT_FORMATS;
265                 for (int i = 0; i < p.length; i++) {
266                     if (i > 0) System.out.print(", ");
267                     System.out.print(p[i].getName());
268                 }
269                 System.out.println("");
270             } else {
271                 m_tripleForm = RDFFormat.forName(rest);
272                 System.out.println("New Triple Result Format: " + m_tripleForm.getName());
273             }
274         } else if (cmd.startsWith("tuplef")) {
275             if (rest.equals("")) {
276                 System.out.println("Current Tuple Result Format: " + m_tupleForm.getName());
277                 System.out.print("Formats Supported: ");
278                 RDFFormat[] p = TupleIterator.OUTPUT_FORMATS;
279                 for (int i = 0; i < p.length; i++) {
280                     if (i > 0) System.out.print(", ");
281                     System.out.print(p[i].getName());
282                 }
283                 System.out.println("");
284             } else {
285                 m_tupleForm = RDFFormat.forName(rest);
286                 System.out.println("New Tuple Result Format: " + m_tupleForm.getName());
287             }
288         } else if (cmd.equals("load")) {
289             checkProfile();
290             doLoad(rest);
291         } else if (cmd.equals("close")) {
292             checkProfile();
293             System.out.print("Closing...");
294             m_connector.close();
295             System.out.println("OK.");
296             m_connector = null;
297             m_profile = null;
298         } else if (cmd.equals("limit")) {
299             doLimit(rest);
300         } else if (cmd.equals("distinct")) {
301             if (m_distinct) {
302                 m_distinct = false;
303                 System.out.println("Forced distinct mode is off.");
304             } else {
305                 m_distinct = true;
306                 System.out.println("Forced distinct mode is on.");
307             }
308         } else if (cmd.equals("trace")) {
309             if (m_trace) {
310                 m_trace = false;
311                 System.out.println("Tracing is disabled.");
312             } else {
313                 m_trace = true;
314                 System.out.println("Tracing is enabled.");
315             }
316         } else {
317             throw new TrippiException("Unrecognized command: " + cmd);
318         }
319     }

321     private static String doAliasReplacements(String string, Map<String, String> aliasMap) {
322         String out = string;
323         Iterator<String> iter = aliasMap.keySet().iterator();
324         while (iter.hasNext()) {
325             String alias = iter.next();
326             String fullForm = aliasMap.get(alias);
327             out = out.replaceAll("<" + alias + ":", "<" + fullForm);
328             out = out.replaceAll("\\^\\^" + alias + ":", "^^" + fullForm);
329         }
330         if (!string.equals(out)) {
331             logger.info("Substituted aliases, string is now: " + out);
332         }
333         return out;
334     }

385     public void doAlias(String rest) throws Exception {
386         Map<String, String> aliases = m_config.getAliasMap();
387         if (rest.equals("")) {
388             // show value of all aliases
389             Iterator<String> iter = aliases.keySet().iterator();
390             while (iter.hasNext()) {
391                 String alias = iter.next();
392                 String value = aliases.get(alias);
393                 System.out.println(alias + "\t" + value);
394             }
395         } else if (rest.equals("!")) {
396             // clear all aliases
397             Map<String, String> emptyMap = new HashMap<String, String>();
398             m_config.setAliasMap(emptyMap);
399             if (m_connector != null) 
400                     m_connector.getReader().setAliasMap(emptyMap);
401             System.out.println("Cleared all aliases.");
402         } else if (rest.indexOf(" ") == -1) {
403             // show value of one alias
404             String val = (String) aliases.get(rest);
405             if (val == null) {
406                 throw new TrippiException("No such alias: " + rest);
407             } else {
408                 System.out.println(rest + "\t" + val);
409             }
410         } else {
411             String[] tokens = rest.split(" ");
412             if (tokens.length != 2) {
413             } else if (tokens[1].equals("!")) {
414                 // clear one alias
415                 aliases.remove(tokens[0]);
416                 m_config.setAliasMap(aliases);
417                 if (m_connector != null) 
418                         m_connector.getReader().setAliasMap(aliases);
419                 System.out.println("Cleared alias: " + tokens[0]);
420             } else {
421                 // set an alias
422                 aliases.put(tokens[0], tokens[1]);
423                 m_config.setAliasMap(aliases);
424                 if (m_connector != null) 
425                         m_connector.getReader().setAliasMap(aliases);
426                 System.out.println(tokens[0] + " is now an alias for " + tokens[1]);
427             }
428         }
429     }

459     public void doTuples(String parms, boolean count) throws Exception {
460         // parse it first
461         String[] tokens = parms.split(" ");
462         if (tokens.length < 2) {
463             throw new TrippiException("No query specified.");
464         }
465         String lang = tokens[0];
466         String query = parms.substring(parms.indexOf(" ")).trim();
467         TupleIterator iter = null;
468         try {
469             long startTime = System.currentTimeMillis();
470             iter = m_connector.getReader().findTuples(lang, query, m_limit, m_distinct);
471             int n = 0;
472             if (count) {
473                 n = iter.count();
474             } else {
475                 iter.setAliasMap(m_connector.getReader().getAliasMap());
476                 n = iter.toStream(System.out, m_tupleForm);
477             }
478             System.out.println("Total Tuples   : " + n);
479             long endTime = System.currentTimeMillis();
480             double elapsedSeconds = ((double) endTime - startTime) / 1000.0;
481             System.out.println("Seconds Elapsed: " + elapsedSeconds);
482         } finally {
483             if (iter != null) iter.close();
484         }
485     }

487     public void doTriples(String parms, boolean count) throws Exception {
488         TripleIterator iter = null;
489         try {
490             int rightPos = parms.indexOf(")");
491             long startTime = System.currentTimeMillis();
492             if (parms.startsWith("(") && rightPos != -1) {
493                 // do a tuples-to-triples query
494                 String pattern = parms.substring(0, rightPos);
495                 String rest = parms.substring(rightPos+1).trim();
496                 String[] t = rest.split(" ");
497                 if (t.length < 2) {
498                     throw new TrippiException("Bad syntax, need lang and query.");
499                 }
500                 String lang = t[0];
501                 String query = rest.substring(rest.indexOf(" ")).trim();
502                 iter = m_connector.getReader().findTriples(lang, query, pattern, m_limit, m_distinct);
503             } else {
504                 // do a real triple query (lang query)
505                 String[] tokens = parms.split(" ");
506                 if (tokens.length < 2) {
507                     throw new TrippiException("Triple query must have at least two parameters.");
508                 }
509                 String lang = tokens[0];
510                 String query = parms.substring(parms.indexOf(" ")).trim();
511                 iter = m_connector.getReader().findTriples(lang, query, m_limit, m_distinct);
512             }
513             doTriples(iter, count, startTime);
514         } finally {
515             if (iter != null) iter.close();
516         }

536     public void doCreate() throws Exception {
537         System.out.println("Creating a new profile. To cancel at any time, enter CANCEL.\n");
538         boolean done = false;
539         boolean canceled = false;
540         String className = null;
541         ConnectorDescriptor descriptor = null;
542         while (!done) {
543             System.out.print("Enter the connector class name: ");
544             className = m_reader.readLine();
545             if (className == null || className.equalsIgnoreCase("CANCEL")) {
546                 canceled = true; done = true;
547             } else if (!className.equals("")) {
548                 try {
549                     descriptor = TriplestoreConnector.getDescriptor(className);
550                     done = true;
551                 } catch (TrippiException e) {
552                     System.out.println("ERROR: " + e.getMessage());
553                 }
554             }
555         }
556         if (canceled) {
557             System.out.println("Canceled.");
558         } else {
559             String id = null;
560             done = false; canceled = false;
561             while (!done) {
562                 System.out.print("Enter a short id for the new profile: ");
563                 id = m_reader.readLine();
564                 if (id == null || id.equalsIgnoreCase("CANCEL")) {
565                     canceled = true; done = true;
566                 } else if (!id.equals("")) {
567                     TrippiProfile p = (TrippiProfile) m_config.getProfiles().get(id);
568                     if (p != null) {
569                         System.out.println("ERROR: A profile with that id already exists.");
570                     } else {
571                         done = true;
572                     }
573                 }
574             }
575             if (canceled) {
576                 System.out.println("Canceled.");
577             } else {
578                 String label = null;
579                 done = false; canceled = false;
580                 while (!done) {
581                     System.out.print("Enter a label for the new profile: ");
582                     label = m_reader.readLine();
583                     if (label == null || label.equalsIgnoreCase("CANCEL")) {
584                         canceled = true; done = true;
585                     } else if (!label.equals("")) {
586                         done = true;
587                     }
588                 }
589                 if (canceled) {
590                     System.out.println("Canceled.");
591                 } else {
592                     TrippiProfile profile = configure(id, label, className, descriptor);
593                     if (profile != null) {
594                         Map<String, TrippiProfile> profileMap = m_config.getProfiles();
595                         profileMap.put(id, profile);
596                         m_config.setProfiles(profileMap);
597                         System.out.println("\nProfile created.  Type 'use " + id + ";' to use it.");
598                     } else {
599                         System.out.println("Canceled.");
600                     }
601                 }
602             }
603         }
604     }

628     private Map<String, String> configure(ConnectorParameter param) throws IOException {
629         Map<String, String> map = new HashMap<String, String>();
630         System.out.println("");
631         System.out.println("Parameter   : " + param.getName());
632         String desc      = "Description : ";
633         String blank     = "              ";
634         System.out.println(desc + formatString(param.getLabel(),
635                                                79 - desc.length(),
636                                                79,
637                                                desc.length()));
638         if (param.getDescription() != null && !param.getDescription().equals("")) {
639             System.out.println(blank + formatString(param.getDescription(),
640                                                     79 - desc.length(),
641                                                     79,
642                                                     desc.length()));
643         }
644         System.out.println("");
645         List<String> options = param.getOptions();
646         if (options.size() == 0) {
647             boolean done = false;
648             String input = null;
649             while (!done) {
650                 System.out.print("Enter a value");
651                 if (param.isOptional()) {
652                     System.out.print(" (or [ENTER] for none)");
653                 }
654                 System.out.print(": ");
655                 input = m_reader.readLine();
656                 if (input == null || input.equalsIgnoreCase("CANCEL")) return null;
657                 if (input.equals("") && param.isOptional()) {
658                     done = true;
659                 }
660                 if (!input.equals("")) {
661                     done = true;
662                 }
663             }
664             if (!input.equals("")) map.put(param.getName(), input);
665         } else {
666             HashMap<String, List<ConnectorParameter>> o = new HashMap<String, List<ConnectorParameter>>();
667             Iterator<String> iter = options.iterator();
668             StringBuffer optString = new StringBuffer();
669             int i = 0;
670             while (iter.hasNext()) {
671                 i++;
672                 String val = iter.next();
673                 o.put(val, param.getParameters(val));
674                 if (i == options.size()) {
675                     optString.append(" or ");
676                 } else if (i > 1) {
677                     optString.append(", ");
678                 }
679                 optString.append(val);
680             }
681             boolean done = false;
682             String input = null;
683             while (!done) {
684                 System.out.print("Enter " + optString);
685                 if (param.isOptional()) {
686                     System.out.print(" (or [ENTER] for none)");
687                 }
688                 System.out.print(": ");
689                 input = m_reader.readLine();
690                 if (input == null || input.equalsIgnoreCase("CANCEL")) return null;
691                 if (input.equals("") && param.isOptional()) {
692                     done = true;
693                 } else if (o.containsKey(input)) {
694                     done = true;
695                 }
696             }
697             if (!input.equals("")) {
698                 // input was ok, so set value in map and move on
699                 map.put(param.getName(), input);
700                 List<ConnectorParameter> subs = o.get(input);
701                 if (subs != null) {
702                     Map<String, String> subMap = configure(subs);
703                     if (subMap == null) return null;
704                     map.putAll(subMap);
705                 }
706             }
707         }
708         return map;
709     }

789     public static String formatString(String in, int firstMax, int otherMax, int otherIndent) {
790         StringBuffer out = new StringBuffer();
791         String[] tokens = in.split(" ");
792         int charsLeft = firstMax;
793         int i = 0;
794         while ( i < tokens.length ) {
795             String token = tokens[i++];
796             out.append(token);
797             charsLeft -= token.length();
798             if (i < tokens.length && tokens[i].length() >= charsLeft) {
799                 out.append('\n');
800                 SpaceCharacters.indent(otherIndent, out);
801                 charsLeft = otherMax - otherIndent;
802             } else {
803                 out.append(' ');
804                 charsLeft--;
805             }
806         }
807         return out.toString();
808     }
