8   /**
9    * A convenience class for getting and printing indent
10   * characters without creating lots of Strings or char arrays.
11   * Not intended for use outside Trippi project.
12   */
13      private static final int SIXTY_FOUR = 64;
14      
15      private static final char[] SIXTY_FOUR_SPACES =
16              indentChars(SIXTY_FOUR);
17      
18      public static char[] indentChars(int num) {
19          char[] indent = new char[num];
20          Arrays.fill(indent, ' ');
21          return indent;
22      }
23      
24      /**
25       * Write a number of whitespaces to a writer
26       * @param num
27       * @param out
28       */
29      public static void indent(int num, PrintWriter out) {
30          if (num <= SIXTY_FOUR) {
31              out.write(SIXTY_FOUR_SPACES, 0, num);
32              return;
33          } else if (num <= 128){
34              out.write(SIXTY_FOUR_SPACES, 0, SIXTY_FOUR);
35              out.write(SIXTY_FOUR_SPACES, 0, num - SIXTY_FOUR);
36          } else {
37              int times = num / SIXTY_FOUR;
38              int rem = num % SIXTY_FOUR;
39              for (int i = 0; i< times; i++) {
40                  out.write(SIXTY_FOUR_SPACES, 0, SIXTY_FOUR);
41              }
42              out.write(SIXTY_FOUR_SPACES, 0, rem);
43              return;
44          }
45      }
46      
47      /**
48       * Write a number of whitespaces to a StringBuffer
49       * @param num
50       * @param out
51       */
52      public static void indent(int num, StringBuffer out) {
53          if (num <= SIXTY_FOUR) {
54              out.append(SIXTY_FOUR_SPACES, 0, num);
55              return;
56          } else if (num <= 128){
57              // avoid initializing loop counters if only one iteration
58              out.append(SIXTY_FOUR_SPACES, 0, SIXTY_FOUR);
59              out.append(SIXTY_FOUR_SPACES, 0, num - SIXTY_FOUR);
60          } else {
61              int times = num / SIXTY_FOUR;
62              int rem = num % SIXTY_FOUR;
63              for (int i = 0; i< times; i++) {
64                  out.append(SIXTY_FOUR_SPACES, 0, SIXTY_FOUR);
65              }
66              out.append(SIXTY_FOUR_SPACES, 0, rem);
67              return;
68          }
