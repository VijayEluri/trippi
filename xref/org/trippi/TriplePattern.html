30      public TriplePattern(Object subject,
31                           Object predicate,
32                           Object object) throws TrippiException {
33          if ( !(subject instanceof String) &&

35              throw new TrippiException("Error in triple pattern: subject "
36                      + "must be a String or a SubjectNode (not " 
37                      + subject.getClass().getName() + ")");

39          if ( !(predicate instanceof String) &&
40                  !(predicate instanceof PredicateNode) ) {
41              throw new TrippiException("Error in triple pattern: predicate "

44          }
45          if ( !(object instanceof String) &&
46                  !(object instanceof ObjectNode) ) {
47              throw new TrippiException("Error in triple pattern: object "
48                      + "must be a String or an ObjectNode (not "

56      public static TriplePattern[] parse(String patterns) 
57                                           throws TrippiException {
58          try {
59              List<String> tokens = tokenize(patterns.replaceAll("\r", " "));
60              if (tokens.size() % 3 != 0 || tokens.size() == 0) {
61                  throw new TrippiException("Triple pattern token count not divisible by 3.");
62              }
63              TriplePattern[] out = new TriplePattern[tokens.size() / 3];
64              Iterator<String> iter = tokens.iterator();
65              int i = 0;
66              RDFUtil factory = new RDFUtil();
67              while (iter.hasNext()) {
68                  Object subject = parseToken(iter.next(), factory);
69                  Object predicate = parseToken(iter.next(), factory);
70                  Object object = parseToken(iter.next(), factory);
71                  out[i++] = new TriplePattern(subject, predicate, object);
72              }
73              return out;
74          } catch (Exception e) {
75              String msg = "Parse error: " + e.getClass().getName();
76              if (e.getMessage() != null) msg += ": " + e.getMessage();
77              throw new TrippiException(msg, e);
78          }
79      }

81      /**
82       * Parse and return a Set of Triples, treating variable bindings
83       * as anonymous nodes.
84       */
85      public static List<Triple> parse(String triples,
86                               GraphElementFactory factory) throws TrippiException {
87          try {
88          TriplePattern[] patterns = parse(triples);
89          HashMap<String, BlankNode> bNodes = new HashMap<String, BlankNode>();
90          ArrayList<Triple> list = new ArrayList<Triple>(patterns.length);
91          for (int i = 0; i < patterns.length; i++) {
92              Object s = patterns[i].getSubject();
93              Object p = patterns[i].getPredicate();
94              Object o = patterns[i].getObject();
95              SubjectNode subject;
96              PredicateNode predicate;
97              ObjectNode object;
98              if (s instanceof String) {
99                  BlankNode b = bNodes.get(s);
100                 if (b == null) {
101                     b = factory.createResource();
102                     bNodes.put((String)s, b);
103                 }
104                 subject = b;
105             } else {
106                 subject = (SubjectNode) s;
107             }
108             if (p instanceof String) {
109                 throw new TrippiException("Predicate cannot be bNode.");
110             } else {
111                 predicate = (PredicateNode) p;
112             }
113             if (o instanceof String) {
114                 BlankNode b = bNodes.get(o);
115                 if (b == null) {
116                     b = factory.createResource();
117                     bNodes.put((String)o, b);
118                 }
119                 object = b;
120             } else {
121                 object = (ObjectNode) o;
122             }
123             list.add(factory.createTriple(subject, predicate, object));
124         }
125         return list;
126         } catch (Exception e) {
127             throw new TrippiException("Error creating triples from string.", e);
128         }

131     private static Object parseToken(String token,
132                                     RDFUtil factory) 
133                 throws GraphElementFactoryException,
134                        URISyntaxException {
135         char c = token.charAt(0);
136         if ( c == '\'' || c == '"' || c == '<' || c == '_' ) {
137             return factory.parse(token);
138         } else if ( c == '$' || c == '?' ) {
139             return token.substring(1);
140         } else {
141             return token;
142         }
143     }

145     private static List<String> tokenize(String patterns) {
146         List<String> tokens = new ArrayList<String>();
147         boolean inToken = false;
148         boolean inLiteral = false;
149         boolean lastWasEscape = false;
150         boolean inQuotedPart = false;
151         char qChar=' ';
152         StringBuffer token = null;
153         for (int i = 0; i < patterns.length(); i++) {
154             char c = patterns.charAt(i);
155             if (inToken) {
156                 if (inLiteral) {
157                     if (inQuotedPart) {
158                         if (lastWasEscape) {
159                             lastWasEscape = false;
160                             token.append(c);
161                         } else {
162                             if (c == qChar) {
163                                 token.append(c);
164                                 inQuotedPart = false;
165                                 if ( i + 1 < patterns.length() ) {
166                                     // are we out of the literal?
167                                     char n = patterns.charAt(i + 1);
168                                     if (n == '\n' || n == '\t' || n == ' ' || n == ',' || n == '(' || n == ')') {
169                                         inToken = false;
170                                         inLiteral = false;
171                                         tokens.add(token.toString());
172                                     }
173                                 }
174                             } else {
175                                 if (c == '\\') {
176                                     lastWasEscape = true;
177                                 }
178                                 token.append(c);
179                             }
180                         }
181                     } else {
182                         if (c == '\n' || c == '\t' || c == ' ' || c == ',' || c == '(' || c == ')') {
183                             inToken = false;
184                             inLiteral = false;
185                             tokens.add(token.toString());
186                         } else {
187                             token.append(c);
188                         }
189                     }
190                 } else {
191                     if (c == '\n' || c == '\t' || c == ' ' || c == ',' || c == '(' || c == ')') {
192                         inToken = false;
193                         tokens.add(token.toString());
194                     } else {
195                         token.append(c);
196                     }
197                 }
198             } else {
199                 if (c != '\n' && c != '\t' && c != ' ' && c != ',' && c != '(' && c != ')') {
200                     inToken = true;
201                     if (c == '\"' || c == '\'') {
202                         inLiteral = true;
203                         inQuotedPart = true;
204                         lastWasEscape = false;
205                         qChar = c;
206                     }
207                     token = new StringBuffer();
208                     token.append(c);
209                 }
210             }
211             if (inToken == false) token = null;
212         }
213         if (token != null) {
214             tokens.add(token.toString());
215         }
216         return tokens;
217     }

252     /**
253      * Get a Triple that matches this pattern using the provided tuple.
254      *
255      * @return Triple if a Triple can be constructed, or null if any of the
256      *                bound variables in the map are null or wouldn't
257      *                match the required type for the triple.
258      * @throws TrippiException if the Map does not contain a key that
259      *                              matches a binding name in this pattern.
260      */
261     public Triple match(Map<String, Node> tuple) throws TrippiException {
262         SubjectNode subject;
263         PredicateNode predicate;
264         ObjectNode object;
265 
266         try {
267 
268             if (m_subject instanceof String) {
269                 Node n = getMatchingNode((String) m_subject, tuple, 0);
270                 if (n == null) return null;
271                 subject = (SubjectNode) n;
272             } else {
273                 if (m_subject instanceof BlankNode) {
274                     // Get a bNode that is unique to the tuple, but may be
275                     // the same as that for other TriplePatterns being evaluated 
276                     // against the same map
277                     subject = RDFFactories.createResource(
278                                tuple.hashCode() + m_subject.hashCode());
279                 } else {
280                     subject = (SubjectNode) m_subject;
281                 }
282             }
283 
284             if (m_predicate instanceof String) {
285                 Node n = getMatchingNode((String) m_predicate, tuple, 1);
286                 if (n == null) return null;
287                 predicate = (PredicateNode) n;
288             } else {
289                 predicate = (PredicateNode) m_predicate;
290             }
291 
292             if (m_object instanceof String) {
293                 Node n = getMatchingNode((String) m_object, tuple, 2);
294                 if (n == null) return null;
295                 object = (ObjectNode) n;
296             } else {
297                 if (m_object instanceof BlankNode) {
298                     // Get a bNode that is unique to the tuple, but may be
299                     // the same as that for other TriplePatterns being evaluated 
300                     // against the same map
301                     object = RDFFactories.createResource(
302                                  tuple.hashCode() + m_object.hashCode());
303                 } else {
304                     object = (ObjectNode) m_object;
305                 }
306             }
307 
308             return RDFFactories.createTriple(subject, predicate, object);
309 
310         } catch (Exception e) {
311             String msg = e.getClass().getName();
312             if (e.getMessage() != null) msg += ": " + e.getMessage();
313             throw new TrippiException(msg, e);
314         }

323     private Node getMatchingNode(String key, 
324                                  Map<String, Node> tuple,
325                                  int type) throws TrippiException {
326         if (tuple.containsKey(key)) {
327             Node node = tuple.get(key);
328             if (node == null) return null; // null value bound for this tuple
329             if (type == 0) {
330                 if (node instanceof SubjectNode) return node;
331                 return null;
332             } else if (type == 1) {
333                 if (node instanceof PredicateNode) return node;
334                 return null;
335             } else {
336                 if (node instanceof ObjectNode) return node;
337                 return null;
338             }
339         } else {
340             throw new TrippiException("No value named '" + key + "' in tuple.");
341         }
342     }

347         try {
348             if (m_subject instanceof String) {

352                 out.append(' ');
353             }
354             if (m_predicate instanceof String) {
355                 out.append(m_predicate).append(' ');
356             } else {
357                 RDFUtil.encode((Node) m_predicate, out);
358                 out.append(' ');
359             }
360             if (m_object instanceof String) {
361                 out.append(m_object);
362             } else {
363                 RDFUtil.encode((Node) m_object, out);
364                 out.append(' ');
365             }
366         } catch (IOException wonthappen) {}
367         return out.toString();
368     }
