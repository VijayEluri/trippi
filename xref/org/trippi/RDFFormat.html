148     public static RDFFormat forName(String name) throws TrippiException {
149         String s = simplifyName(name);
150         for (int i = 0; i < ALL.length; i++) {
151             if (s.equals(simplifyName(ALL[i].getName()))) return ALL[i];
152         }
153         throw new TrippiException("Unrecognized format: " + name);
154     }
155 
156     private static final String simplifyName(String name) {
157         name = name.toLowerCase()
158                    .trim()
159                    .replaceAll(" ", "")
160                    .replaceAll("-", "")
161                    .replaceAll("/", "")
162                    .replaceAll("-", "");
163         if (name.equals("nt")) return "ntriples";
164         if (name.equals("n3")) return "notation3";
165         if (name.equals("ttl")) return "turtle";
166         if (name.equals("rdf")) return "rdfxml";
167         return name;
168     }
