21  /**
22   * Adapter for using Sesame's RIO RdfDocumentWriters for triple serialization.
23   */
24  public class RIOTripleWriter extends TripleWriter {
25  
26      private RDFWriter m_writer;
27      private org.openrdf.model.impl.ValueFactoryImpl m_rioFactory = 
28              new org.openrdf.model.impl.ValueFactoryImpl();
29  
30      public RIOTripleWriter(RDFWriter writer, Map<String, String> aliases) throws TrippiException {
31          try {
32              m_writer = writer;
33              Iterator<String> iter = aliases.keySet().iterator();
34              while (iter.hasNext()) {
35                  String prefix = iter.next();
36                  String name = aliases.get(prefix);
37                  m_writer.handleNamespace(prefix, name);
38              }
39          } catch (RDFHandlerException e) {
40              throw new TrippiException("Error setting up RIOTripleWriter", e);
41          }
42      }
43  
44      public RIOTripleWriter(RDFWriter writer, AliasManager aliases) throws TrippiException {
45          try {
46              m_writer = writer;
47              if (aliases != null) {
48                  for (Alias alias: aliases.getAliases().values()){
49                      m_writer.handleNamespace(alias.getKey(), alias.getExpansion());
50                  }
51              }
52          } catch (RDFHandlerException e) {
53              throw new TrippiException("Error setting up RIOTripleWriter", e);
54          }
55      }
56  
57      @Override
58  	public int write(TripleIterator iter) throws TrippiException {
59          try {
60              m_writer.startRDF();
61              int count = 0;
62              while (iter.hasNext()) {
63                  Triple triple = iter.next();
64                  // write the triple to the rio writer using rio equivalents
65                  m_writer.handleStatement(new StatementImpl(rioResource(triple.getSubject()),
66                                          rioURI(triple.getPredicate()),
67                                          rioValue(triple.getObject())));
68                  count++;
69              }
70              m_writer.endRDF();
71              iter.close();
72              return count;
73          } catch (RDFHandlerException e) {
74              throw new TrippiException("Error setting up RIOTripleWriter", e);
75          }
76      }

78      private org.openrdf.model.Resource rioResource(SubjectNode s) {
79          if (s instanceof BlankNode) {
80              return m_rioFactory.createBNode("" + s.hashCode());
81          } else {
82              return m_rioFactory.createURI(((URIReference) s).getURI().toString());
83          }
84      }
85  
86      private org.openrdf.model.URI rioURI(PredicateNode p) {
87          return m_rioFactory.createURI(((URIReference) p).getURI().toString());
88      }
89  
90      private org.openrdf.model.Value rioValue(ObjectNode o) {
91          if (o instanceof BlankNode) {
92              return m_rioFactory.createBNode("" + o.hashCode());
93          } else if (o instanceof Literal) {
94              Literal l = (Literal) o;
95              if (l.getDatatypeURI() != null) {
96                  return m_rioFactory.createLiteral(l.getLexicalForm(), 
97                                                    m_rioFactory.createURI(l.getDatatypeURI().toString()));
98              } else if (l.getLanguage() != null && !l.getLanguage().equals("")) {
99                  return m_rioFactory.createLiteral(l.getLexicalForm(), l.getLanguage());
100             } else {
101                 return m_rioFactory.createLiteral(l.getLexicalForm());
102             }
103         } else {
104             return m_rioFactory.createURI(((URIReference) o).getURI().toString());
105         }
106     }
