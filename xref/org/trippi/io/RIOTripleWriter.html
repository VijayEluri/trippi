19  /**
20   * Adapter for using Sesame's RIO RdfDocumentWriters for triple serialization.
21   */
22  public class RIOTripleWriter extends TripleWriter {
23  
24      private RDFWriter m_writer;
25      private org.openrdf.model.impl.ValueFactoryImpl m_rioFactory = 
26              new org.openrdf.model.impl.ValueFactoryImpl();
27  
28      public RIOTripleWriter(RDFWriter writer, Map<String, String> aliases) throws TrippiException {
29          try {
30              m_writer = writer;
31              Iterator<String> iter = aliases.keySet().iterator();
32              while (iter.hasNext()) {
33                  String prefix = iter.next();
34                  String name = aliases.get(prefix);
35                  m_writer.handleNamespace(prefix, name);
36              }
37          } catch (RDFHandlerException e) {
38              throw new TrippiException("Error setting up RIOTripleWriter", e);
39          }
40      }
41  
42      @Override
43  	public int write(TripleIterator iter) throws TrippiException {
44          try {
45              m_writer.startRDF();
46              int count = 0;
47              while (iter.hasNext()) {
48                  Triple triple = iter.next();
49                  // write the triple to the rio writer using rio equivalents
50                  m_writer.handleStatement(new StatementImpl(rioResource(triple.getSubject()),
51                                          rioURI(triple.getPredicate()),
52                                          rioValue(triple.getObject())));
53                  count++;
54              }
55              m_writer.endRDF();
56              iter.close();
57              return count;
58          } catch (RDFHandlerException e) {
59              throw new TrippiException("Error setting up RIOTripleWriter", e);
60          }
61      }
62  
63      private org.openrdf.model.Resource rioResource(SubjectNode s) {
64          if (s instanceof BlankNode) {
65              return m_rioFactory.createBNode("" + s.hashCode());
66          } else {
67              return m_rioFactory.createURI(((URIReference) s).getURI().toString());
68          }
69      }
70  
71      private org.openrdf.model.URI rioURI(PredicateNode p) {
72          return m_rioFactory.createURI(((URIReference) p).getURI().toString());
73      }
74  
75      private org.openrdf.model.Value rioValue(ObjectNode o) {
76          if (o instanceof BlankNode) {
77              return m_rioFactory.createBNode("" + o.hashCode());
78          } else if (o instanceof Literal) {
79              Literal l = (Literal) o;
80              if (l.getDatatypeURI() != null) {
81                  return m_rioFactory.createLiteral(l.getLexicalForm(), 
82                                                    m_rioFactory.createURI(l.getDatatypeURI().toString()));
83              } else if (l.getLanguage() != null && !l.getLanguage().equals("")) {
84                  return m_rioFactory.createLiteral(l.getLexicalForm(), l.getLanguage());
85              } else {
86                  return m_rioFactory.createLiteral(l.getLexicalForm());
87              }
88          } else {
89              return m_rioFactory.createURI(((URIReference) o).getURI().toString());
90          }
91      }
