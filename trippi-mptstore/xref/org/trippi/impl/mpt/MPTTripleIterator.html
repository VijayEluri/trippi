63      private org.jrdf.graph.Node mptToJRDF(Node mptNode) {
64  
65          try {
66              if (mptNode instanceof URIReference) {
67                  URIReference mptURIReference = (URIReference) mptNode;
68                  return _util.createResource(mptURIReference.getURI());
69              } else if (mptNode instanceof Literal) {
70                  Literal mptLiteral = (Literal) mptNode;
71                  if (mptLiteral.getLanguage() != null) {
72                      return _util.createLiteral(mptLiteral.getValue(),
73                                                 mptLiteral.getLanguage());
74                  } else if (mptLiteral.getDatatype() != null) {
75                      return _util.createLiteral(mptLiteral.getValue(),
76                                                 mptLiteral.getDatatype().getURI());
77                  } else {
78                      return _util.createLiteral(mptLiteral.getValue());
79                  }
80              } else {
81                  throw new RuntimeException("Unrecognized node type: " 
82                          + mptNode.getClass().getName());
83              }
84          } catch (GraphElementFactoryException e) {
