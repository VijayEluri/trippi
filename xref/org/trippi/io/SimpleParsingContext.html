23  /**
24   * An iterator over triples parsed by a RIO rdf parser.
25   *
26   * @author armintor@gmail.com
27   */
28  public class SimpleParsingContext<T> implements RDFHandler {
29  
30      private static final Logger logger =
31          LoggerFactory.getLogger(SimpleParsingContext.class.getName());
32  
33      protected int m_tripleCount = 0;
34  
35      protected final HashSet<T> m_triples =
36              new HashSet<T>();
37      
38      private final Transformer<T> m_transform;
39      
40      /**
41       * Initialize the iterator by starting the parsing thread.
42       * @throws IOException 
43       * @throws RDFHandlerException 
44       * @throws RDFParseException 
45       */
46      protected SimpleParsingContext(InputStream in, 
47                  RDFParser parser, 
48                  String baseURI,
49                  Transformer<T> transform)
50          throws TrippiException, RDFParseException, RDFHandlerException, IOException {
51          parser.setRDFHandler(this);
52          parser.setVerifyData(true);
53          parser.setStopAtFirstError(false);
54          this.m_transform = transform;
55          try {
56              parser.parse(in, baseURI);
57          } finally {
58              try {
59                  in.close();
60              } catch (IOException ioe) {
61                  // just logging any additional exceptions
62                  logger.warn(ioe.getMessage());
63              }
64          }
65      }
66      
67      public Set<T> getSet() {
68          return m_triples;
69      }
70      
71      public TrippiIterator<T> getIterator() {
72          return new SimpleTrippiIterator<T>(m_triples);
73      }

76      /**

79       */
80      public void handleNamespace(String prefix, String uri) {
81          // no-op    
82      }
83      
84      public void endRDF() throws RDFHandlerException {
85          // no-op    
86      }
87  
88      public void handleComment(String arg0) throws RDFHandlerException {

90      }
91  
92      public void handleStatement(Statement st)
93              throws RDFHandlerException {
94          // first, convert the rio statement to jrdf nodes
95          m_triples.add(m_transform.transform(st, RDFFactories.FACTORY));
96      }
