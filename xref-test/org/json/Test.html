73      public void testNull() throws Exception {
74          JSONObject jsonobject;
75  
76          jsonobject = new JSONObject("{\"message\":\"null\"}");
77          assertFalse(jsonobject.isNull("message"));
78          assertEquals("null", jsonobject.getString("message"));
79  
80          jsonobject = new JSONObject("{\"message\":null}");
81          assertTrue(jsonobject.isNull("message"));
82          assertEquals(null, jsonobject.getString("message"));
83      }
